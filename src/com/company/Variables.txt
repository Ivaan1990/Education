1. boolean - Логический {true, false} 1-bit
2. byte  - Целочисленный {-128....127} -8 bit [-2^7...2^7-1]
3. short - Целочисленный {-32768(2^15)...32767} - 16 bit
4. int - Целочисленный {-2^23...+2^23-1} - 24 bit -Основная
5. long - Целочисленный {-2^31...+2^31-1} -32 bit
6. char - Символьный - почти как short -16 bit {0-65456}
7. float - для чисел с плавающей запятой [5.45478e8 = 5.45478*10^8=545578000] 128 bit
8. double - для чисел с плавающей запятой 256 bit
